#include<iostream>
using namespace std;

int main()
{
    int n[50],m[20],i,j,k=0;
    for(i=0;i<n;i++)
        cin>>n[i];
    for(j=0;j<n;j++)
        cin>>m[j];
    for(i=k,j=0;i<10,j<5;j++,i++)
    {
        if(n[i]!=m[j])
        {
            k=i;
            j=0;
        }
        else 
            cout<<"match is found :-"<<m;
    return 0;
}
